// CreateTodo.telefunc.ts
// Environment: Node.js server

import { getContext, Abort } from 'telefunc'

// Telefunc makes onNewTodo() remotely callable
// from the browser.
export { onNewTodo }

// Telefunc infers TypeScript types to guarantee that
// `text` is a `string` at runtime.
async function onNewTodo(text: string) {
  const { user } = getContext()
  // Abort if the user isn't logged-in
  if (!user) throw Abort()

  // With an ORM:
  const todoItem = new Todo({text, authorId: user.id})
  await todoItem.save()

  // Or with SQL:
  await execute(
    "INSERT INTO todo_items VALUES (:text, :authorId)",
    { text, authorId: user.id }
  )
}

// CreateTodo.tsx
// Environment: Browser

// CreateTodo.telefunc.ts isn't actually loaded: Tele-
// func transforms `.telefunc.(js|ts)` files into a
// thin HTTP client.
import { onNewTodo } from './CreateTodo.telefunc.ts'

async function onClick(form: Form) {
  const text = form.input.value
  // Behind the scenes, Telefunc makes an HTTP request
  // to our Node.js server.
  await onNewTodo(text)
}

function CreateTodo() {
  return (
    <form>
      <input input="text"></input>
      <button onClick={onClick}>Add To-Do</button>
    </form>
  )
}
